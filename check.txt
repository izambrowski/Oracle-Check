SELECT TICKER,EXCHANGE,FIRST_COMPARABLE_PUBLISH_UUID,cf.COMP_FILE_FISC_YR_NUM,fp.FISC_PRD_DESC,TASK_ID,TASK_TYPE_NM
FROM UWF.TASK_SUMMARY ts
JOIN UWF.COMP_FILE cf ON CONCAT(cf.BB_ID,cf.TIME_BLOCK_INSTANCE_ID) = CONCAT(ts.BB_ID,ts.TIME_BLOCK_INSTANCE_ID)
JOIN UWF.PERIOD_KEY pk ON cf.PERIOD_KEY_ID = pk.PERIOD_KEY_ID
JOIN UWF.FISCAL_PERIOD fp ON fp.FISC_PRD_NUM = pk.FISC_PRD_NUM
WHERE FIRST_COMPARABLE_PUBLISH_UUID is NOT NULL
AND COMP_FILE_TYPE_NUM = 8
AND FULL_TICKER = 'MCD US'
AND cf.COMP_FILE_FISC_YR_NUM = 2018
AND fp.FISC_PRD_DESC = 'Quarter 2'
